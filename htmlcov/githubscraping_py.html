10# 

11# A GITHUB CRAWLER THAT IMPLEMENTS THE GITHUB SEARCH AND RETURNS ALL THE 

12# LINKS FROM THE SEARCH RESULT. 

13# 

14# Usage: 

15# from githubscraping import scrapeMain 

16# 

17# output = scrapeMain('/path/to/input/json', '/path/to/output/json') 

18# # print(output)

20# 

21# Input must be a json file of the form: 

22# { 

23# "keywords": [ 

24# "openstack", 

25# "nova", 

26# "css" 

27# ], 

28# "proxies": [ 

29# "202.14.80.2:3128", 

30# "116.203.23.252:3128" 

31# ], 

32# "type": "Repositories" 

33# } 

34# 

35# Output will be a json file with a list of urls to the found repositories. 

36# When searching for repositories it will also have the extra data of owner

51def jsonInput(input_file): 

52 ''' Parse input data into a dict ''' 

53 with open(input_file, 'r') as file: 

54 data = json.load(file) 

55 data['type'] = data['type'].lower() 

56 

57 return data 

58 

59def jsonOutput(output_file, json_string): 

60 ''' Save output to json ''' 

61 with open(output_file, 'w') as file: 

62 file.write(json_string) 

63 

64def urlCreate(query, se_type): 

65 ''' Create URL from the keywords and the type ''' 

66 # Create string from list of queried items 

67 query_str = '+'.join(query) 

68 # Create URL 

69 url = 'https://github.com/search?q=' + query_str + '&type=' + se_type 

70 

71 return url

73def proxier(proxy_lst): 

74 ''' 

75 Select a random proxy from the list of provided proxies. 

76 Create the structure that requests uses for proxies. 

77 Return None if no proxy. 

78 ''' 

79 if proxy_lst: 

80 # Get random item from the list 

81 rnd_proxy = random.choice(proxy_lst) 

82 

83 # Request recent versions bug: 

84 # Need to add http:// or https:// to the proxies to work 

85 proxy = { 

86 'http': 'http://' + rnd_proxy, 

87 'https': 'http://' + rnd_proxy 

88 } 

89 # If list empty/no list: proxy is None 

90 else: 

91 proxy = None 

92 

93 return proxy

95def searchScrapper(url, se_type, proxy_lst): 

96 # Select proxy 

97 proxy = proxier(proxy_lst) 

98 

99 # Access the page 

100 rsp=requests.get(url, proxies=proxy, stream=True) 

101 

102 # IP used For Debugging, to make sure youre using a proxy 

103 #ip_msg = 'IP used: ' + str(rsp.raw._connection.sock.getpeername()) 

104 #print(ip_msg) 

105 

106 # Proccess the html 

107 soup = BeautifulSoup(rsp.content, 'html.parser') 

108 

109 # Different searches for different types, since the returning list items 

110 # don't use the same structure 

111 # Repositories 

112 if se_type == 'repositories': 

113 searched = soup.find('ul', attrs={'class': 'repo-list'}) 

114 items = searched.find_all('a', attrs={'class': 'v-align-middle'}) 

115 # Issues 

116 elif se_type == 'issues': 

117 searched = soup.find('div', attrs={'class': 'issue-list'}) 

118 items = searched.find_all('a', attrs={'class': 'Link--muted text-bold'}) 

119 # Wikis 

120 elif se_type == 'wikis': 

121 searched = soup.find('div', attrs={'id': 'wiki_search_results'}) 

122 divs = searched.find_all('div', attrs={'class': 'f4 text-normal'}) 

123 items = [] 

124 for div in divs: 

125 items.append(div.find('a')) 

126 

127 # Loop the 'a' tags to get the urls 

128 to_json_list = [] 

129 for item in items: 

130 got_url = item.get('href') 

131 # Format URL 

132 full_url = 'https://github.com' + got_url 

133 

134 # Extra Information when searching repositories 

135 if se_type == 'repositories': 

136 # Get repository owner 

137 owner = got_url.split('/')[1] 

138 # Get language stats 

139 lan_stats = statScrapper(full_url, proxy) 

140 # Append to list for json output 

141 to_json_list.append({'url': full_url, 

142 'extra': {'owner': owner, 

143 'language_stats': lan_stats} 

144 }) 

145 else: 

146 # Append to list for json output 

147 to_json_list.append({'url': full_url}) 

148 

149 json_string = json.dumps(to_json_list, indent=4) 

150 

151 return json_string

153def statScrapper(url, proxy): 

154 ''' Get the extra information about language percentages used in repo ''' 

155 # Request 

156 rsp=requests.get(url, proxies=proxy) 

157 # Parse html 

158 soup = BeautifulSoup(rsp.content, 'html.parser') 

159 

160 # Find the languages bar 

161 spans = soup.find_all('span', attrs={'class': 'Progress-item', 

162 'data-view-component': "true"} 

163 ) 

164 stat_dict = {} 

165 for span in spans: 

166 try: 

167 # Get the item value and split it into vars 

168 lang, percent = span['aria-label'].split(' ') 

169 # Save to dict 

170 # String to float because the target example had it as float 

171 stat_dict[lang] = float(percent) 

172 # For 'span' that dont have that item (what we looking for), pass 

173 except KeyError: 

174 pass 

175 

176 return stat_dict

178def scrapeMain(input_json, output_json): 

179 ''' 

180 Main Function, arguments input and output json file path. Easy of use: 

181 * Takes input json and parsesd the data 

182 * Creates the url from the input 

183 * Does the actual scraping 

184 * Saves to output json 

185 * Return the json string 

186 ''' 

187 # Open input json 

188 data = jsonInput(input_json) 

189 # Create URL 

190 url = urlCreate(data['keywords'], data['type']) 

191 # Call to Scrape 

192 json_string = searchScrapper(url, data['type'], data['proxies']) 

193 # Save output 

194 jsonOutput(output_json, json_string) 

195 

196 return json_string
